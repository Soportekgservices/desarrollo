FUNCTION,actualizar_tvalores_facturacion,"p_id_distribuidor uuid, p_id_colegio integer","CREATE OR REPLACE FUNCTION public.actualizar_tvalores_facturacion(p_id_distribuidor uuid, p_id_colegio integer)
 RETURNS jsonb
 LANGUAGE plpgsql
 SECURITY DEFINER
 SET search_path TO 'public'
AS $function$
DECLARE
    v_tot_estudiantes INTEGER;
    v_tot_acumulado NUMERIC;
    v_resultado JSONB;
BEGIN
    
    -- 1️⃣ CONTAR ESTUDIANTES ACTIVOS DEL DISTRIBUIDOR EN EL COLEGIO
    SELECT COUNT(DISTINCT e.id)
    INTO v_tot_estudiantes
    FROM public.testudiantes e
    JOIN public.tcolegios c ON e.id_colegio = c.id
    WHERE c.id_dist = p_id_distribuidor
        AND c.id = p_id_colegio
        AND e.estado = 'Activo';
    
    -- 2️⃣ SUMAR TOTAL DE FACTURAS NO CANCELADAS (ingresos pendientes)
    SELECT COALESCE(SUM(total), 0)
    INTO v_tot_acumulado
    FROM public.tfacturas
    WHERE id_distribuidor = p_id_distribuidor
        AND id_colegio = p_id_colegio
        AND pagada = FALSE;
    
    -- 3️⃣ UPSERT en tvalores (Insert or Update)
    INSERT INTO public.tvalores (
        id_distribuidor,
        id_colegio,
        tot_estcolegios,
        tot_acumpesos
    )
    VALUES (
        p_id_distribuidor,
        p_id_colegio,
        v_tot_estudiantes,
        v_tot_acumulado
    )
    ON CONFLICT (id_distribuidor, id_colegio) DO UPDATE SET
        tot_estcolegios = EXCLUDED.tot_estcolegios,
        tot_acumpesos = EXCLUDED.tot_acumpesos;
    
    v_resultado := jsonb_build_object(
        'status', 'success',
        'id_distribuidor', p_id_distribuidor,
        'id_colegio', p_id_colegio,
        'estudiantes_activos', v_tot_estudiantes,
        'ingresos_pendientes', v_tot_acumulado
    );
    
    RETURN v_resultado;
    
EXCEPTION WHEN OTHERS THEN
    RETURN jsonb_build_object(
        'status', 'error',
        'message', SQLERRM,
        'code', SQLSTATE
    );
END;
$function$

 RETURNS jsonb
 LANGUAGE plpgsql
 SECURITY DEFINER
AS $function$
DECLARE
    v_factura record;
    v_subtotal numeric(12,2);
    v_impuesto numeric(12,2);
    v_total numeric(12,2);
    v_antes jsonb;
    v_despues jsonb;
BEGIN
    -- Bloquear y verificar existencia de la factura
    SELECT * INTO v_factura FROM public.tfacturas WHERE id = p_factura_id FOR UPDATE;
    IF NOT FOUND THEN
        RETURN jsonb_build_object('status', 'error', 'message', 'Factura no encontrada');
    END IF;
    -- Registrar instantánea del estado anterior
    v_antes := jsonb_build_object(
        'cantidad_estudiantes', v_factura.cantidad_estudiantes,
        'subtotal', v_factura.subtotal,

        'precio_unitario_real', v_factura.precio_unitario_real
    );
    -- Calcular nuevos montos financieros
    v_subtotal := ROUND((COALESCE(v_factura.precio_base_at_emission, 0) + (p_cant_real * p_precio_real))::numeric, 2);
    
    -- Validaciones preventivas para evitar violación de chk_facturas_montos

    END IF;
    v_impuesto := ROUND(((v_subtotal - p_descuento_real) * (COALESCE(v_factura.impuesto_porcentaje_at_emission, 0) / 100.0))::numeric, 2);
    v_total := ROUND(((v_subtotal - p_descuento_real) + v_impuesto)::numeric, 2);
    IF v_total < 0 THEN
        RETURN jsonb_build_object('status', 'error', 'message', 'El total calculado de la factura no puede ser negativo ($' || v_total || ').');
    END IF;
    -- Actualizar factura conservando las columnas heredadas
    UPDATE public.tfacturas
    SET
        cantidad_estudiantes_real = p_cant_real,
        precio_unitario_real = p_precio_real,
        subtotal_real = v_subtotal,
        descuento_real = p_descuento_real,
        impuesto_real = v_impuesto,
        total_real = v_total,
        -- Sincronizar columnas legacy
        cantidad_estudiantes = p_cant_real,
        subtotal = v_subtotal,
        descuento = p_descuento_real,
        impuesto = v_impuesto,
        total = v_total,
        notas_admin = p_notas,
        updated_at = NOW()
    WHERE id = p_factura_id;
    -- Registrar instantánea del nuevo estado
    v_despues := jsonb_build_object(
        'cantidad_estudiantes', p_cant_real,
        'subtotal', v_subtotal,
        'descuento', p_descuento_real,
        'impuesto', v_impuesto,
        'total', v_total,
        'cantidad_estudiantes_real', p_cant_real,
        'precio_unitario_real', p_precio_real
    );

        'subtotal', v_subtotal,

    -- 0️⃣ Depuración: imprimir el token recibido y su longitud

    RAISE NOTICE 'Token recibido en login_unificado: % (len=%)',

FUNCTION,obtener_examen_estudiante_seguro,"p_estudiante_id uuid, p_grupo_id integer","CREATE OR REPLACE FUNCTION public.obtener_examen_estudiante_seguro(p_estudiante_id uuid, p_grupo_id integer)
 RETURNS jsonb
 LANGUAGE plpgsql
 SECURITY DEFINER
 SET search_path TO 'public'
AS $function$
DECLARE
    v_ahora_col  timestamp := NOW() AT TIME ZONE 'America/Bogota';
    v_hoy        date      := v_ahora_col::date;
    v_hora       time      := v_ahora_col::time;
    v_resultado_previo RECORD;
    v_solicitud RECORD;
    v_preguntas jsonb;
    v_pend_count integer;
BEGIN
    SELECT id, estado, respuestas_raw INTO v_resultado_previo
    FROM public.tresultados
    WHERE id_estudiante = p_estudiante_id;

    SELECT sa.*, row_to_json(p.*) as metadata_prueba
    INTO v_solicitud
    FROM public.tsolicitudes_aplicacion sa
    JOIN public.tpruebas p ON sa.id_prueba = p.id
    WHERE sa.id_grupo = p_grupo_id
      AND sa.estado = 'aprobada'
      AND v_hoy >= sa.fecha_inicio
      AND v_hoy <= sa.fecha_fin
      AND (
          -- Sin franja horaria: acceso todo el día
          sa.hora_inicio IS NULL
          OR (
              -- Mismo día: hora actual dentro del rango
              sa.fecha_inicio = sa.fecha_fin
              AND v_hora >= sa.hora_inicio
              AND v_hora <= sa.hora_fin
          )
          OR (
              -- Días distintos (cross-day):
              sa.fecha_inicio != sa.fecha_fin
              AND (
                  (v_hoy = sa.fecha_inicio AND v_hora >= sa.hora_inicio)
                  OR (v_hoy > sa.fecha_inicio AND v_hoy < sa.fecha_fin)
                  OR (v_hoy = sa.fecha_fin AND v_hora <= sa.hora_fin)
              )
          )
      )
    ORDER BY sa.id DESC
    LIMIT 1;

    IF v_solicitud.id IS NOT NULL THEN
        SELECT jsonb_agg(q ORDER BY num_pregunta ASC) INTO v_preguntas
        FROM public.tpreguntas q
        WHERE q.id_prueba = v_solicitud.id_prueba;
    END IF;

    SELECT count(*)::integer INTO v_pend_count
    FROM public.tsolicitudes_aplicacion
    WHERE id_grupo = p_grupo_id AND estado = 'pendiente';

    RETURN jsonb_build_object(
        'status', 'success',
        'resultado_previo', row_to_json(v_resultado_previo),
        'solicitud_activa', row_to_json(v_solicitud),
        'preguntas', v_preguntas,
        'tiene_pendientes', (v_pend_count > 0)
    );
EXCEPTION WHEN OTHERS THEN
    RETURN jsonb_build_object('status', 'error', 'message', SQLERRM);
END;

    v_abonos_tot numeric(12,2);
    v_abono_id bigint;
    v_pagada_final boolean;

    SELECT * INTO v_factura FROM public.tfacturas WHERE id = p_factura_id FOR UPDATE;
    IF NOT FOUND THEN
        RETURN jsonb_build_object('status', 'error', 'message', 'Factura no encontrada');
    END IF;
    -- Inicializar factura si es legacy (total = 0)
    IF COALESCE(v_factura.total, 0) = 0 THEN
        DECLARE
            v_sub_init numeric(12,2);
            v_tax_init numeric(12,2);
            v_tot_init numeric(12,2);
            v_cant_init integer;
            v_punit_init numeric(12,2);

            v_cant_init := COALESCE(
                v_factura.cantidad_estudiantes_completaron,
                (SELECT COUNT(*)::integer FROM public.tcompletados_pruebas WHERE id_solicitud = v_factura.id_solicitud_origen),
                0
            );
            v_punit_init := COALESCE(
                v_factura.precio_unitario_calculado,
                (SELECT precio_por_estudiante FROM public.tprecios WHERE estado = 'Activo' ORDER BY created_at DESC LIMIT 1),
                0
            );
            v_sub_init := ROUND((COALESCE(v_factura.precio_base_at_emission, 0) + (v_cant_init * v_punit_init))::numeric, 2);
            v_tax_init := ROUND(((v_sub_init - COALESCE(v_factura.descuento, 0)) * (COALESCE(v_factura.impuesto_porcentaje_at_emission, 8) / 100.0))::numeric, 2);
            v_tot_init := ROUND(((v_sub_init - COALESCE(v_factura.descuento, 0)) + v_tax_init)::numeric, 2);
            
            UPDATE public.tfacturas
            SET 
                subtotal = v_sub_init,
                total = v_tot_init,
                impuesto = v_tax_init,
                cantidad_estudiantes = v_cant_init,
                precio_unitario_calculado = v_punit_init,
                subtotal_real = v_sub_init,
                total_real = v_tot_init,
                impuesto_real = v_tax_init,
                cantidad_estudiantes_real = v_cant_init,
                precio_unitario_real = v_punit_init,
                updated_at = NOW()
            WHERE id = p_factura_id;
            
            -- Volver a leer la factura con los valores cargados
            SELECT * INTO v_factura FROM public.tfacturas WHERE id = p_factura_id;
        END;
    END IF;
    -- Registrar instantánea previa

        'fecha_pago', v_factura.fecha_pago,
        'referencia_pago', v_factura.referencia_pago

    FROM public.tabonos_factura
    WHERE id_factura = p_factura_id AND estado = 'Aplicado';

    ) VALUES (
        p_factura_id, p_monto, p_metodo, p_referencia, 'Aplicado', p_usuario_id
    ) RETURNING id INTO v_abono_id;
    -- Recalcular saldo total de abonos activos

        'pagada', v_pagada_final,
        'fecha_pago', CASE WHEN v_pagada_final THEN CURRENT_DATE ELSE NULL END,
