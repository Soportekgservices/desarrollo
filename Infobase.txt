FUNCTION,actualizar_tvalores_facturacion,"p_id_distribuidor uuid, p_id_colegio integer","CREATE OR REPLACE FUNCTION public.actualizar_tvalores_facturacion(p_id_distribuidor uuid, p_id_colegio integer)
 RETURNS jsonb
 LANGUAGE plpgsql
 SECURITY DEFINER
 SET search_path TO 'public'
AS $function$
DECLARE
    v_tot_estudiantes INTEGER;
    v_tot_acumulado NUMERIC;
    v_resultado JSONB;
BEGIN
    
    -- 1️⃣ CONTAR ESTUDIANTES ACTIVOS DEL DISTRIBUIDOR EN EL COLEGIO
    SELECT COUNT(DISTINCT e.id)
    INTO v_tot_estudiantes
    FROM public.testudiantes e
    JOIN public.tcolegios c ON e.id_colegio = c.id
    WHERE c.id_dist = p_id_distribuidor
        AND c.id = p_id_colegio
        AND e.estado = 'Activo';
    
    -- 2️⃣ SUMAR TOTAL DE FACTURAS NO CANCELADAS (ingresos pendientes)
    SELECT COALESCE(SUM(total), 0)
    INTO v_tot_acumulado
    FROM public.tfacturas
    WHERE id_distribuidor = p_id_distribuidor
        AND id_colegio = p_id_colegio
        AND pagada = FALSE;
    
    -- 3️⃣ UPSERT en tvalores (Insert or Update)
    INSERT INTO public.tvalores (
        id_distribuidor,
        id_colegio,
        tot_estcolegios,
        tot_acumpesos
    )
    VALUES (
        p_id_distribuidor,
        p_id_colegio,
        v_tot_estudiantes,
        v_tot_acumulado
    )
    ON CONFLICT (id_distribuidor, id_colegio) DO UPDATE SET
        tot_estcolegios = EXCLUDED.tot_estcolegios,
        tot_acumpesos = EXCLUDED.tot_acumpesos;
    
    v_resultado := jsonb_build_object(
        'status', 'success',
        'id_distribuidor', p_id_distribuidor,
        'id_colegio', p_id_colegio,
        'estudiantes_activos', v_tot_estudiantes,
        'ingresos_pendientes', v_tot_acumulado
    );
    
    RETURN v_resultado;
    
EXCEPTION WHEN OTHERS THEN
    RETURN jsonb_build_object(
        'status', 'error',
        'message', SQLERRM,
        'code', SQLSTATE
    );
END;

 RETURNS json
 LANGUAGE plpgsql
 SECURITY DEFINER
 SET search_path TO 'public'
AS $function$
DECLARE
    r jsonb; -- CAMBIO CLAVE: Declarado como jsonb para permitir el uso de ->>
    v_procesados integer := 0;
    v_user_id uuid;
    v_grado_id integer;
    v_grupo_id integer;
    v_tipodoc_limpio text;
BEGIN
    -- Iteramos obteniendo los elementos jsonb directamente
    FOR r IN SELECT jsonb_array_elements(p_estudiantes)
    LOOP
        -- 1. Limpieza y Normalización del Tipo de Documento
        v_tipodoc_limpio := UPPER(REPLACE(COALESCE(r->>'tipodoc', ''), '.', ''));
        IF v_tipodoc_limpio = 'CC' THEN v_tipodoc_limpio := 'C.C'; END IF;
        IF v_tipodoc_limpio = 'CE' THEN v_tipodoc_limpio := 'C.E'; END IF;

        IF (r->>'nombre' IS NULL OR r->>'identificacion' IS NULL OR v_tipodoc_limpio NOT IN ('C.C', 'TI', 'C.E', 'EXT', 'PPT')) THEN
            CONTINUE;
        END IF;

        SELECT id INTO v_grado_id FROM public.tgrados WHERE nombre = (r->>'grado') LIMIT 1;
        IF NOT FOUND THEN
            INSERT INTO public.tgrados (nombre, numero) 
            VALUES (r->>'grado', CASE WHEN (r->>'grado') ~ '^[0-9]+$' THEN (r->>'grado')::integer ELSE 0 END) 
            RETURNING id INTO v_grado_id;
        END IF;

        SELECT id INTO v_grupo_id FROM public.tgrupos 
        WHERE nombre = (r->>'grupo') AND id_colegio = p_colegio_id AND id_grado = v_grado_id

        -- 5. Sincronización en TUSUARIO (Identidad raíz)
        INSERT INTO public.tusuario (nombre, identificacion, tipodoc, rol, estado)
        VALUES (
            r->>'nombre',
            r->>'identificacion',
            v_tipodoc_limpio::tipo_documento_es,
            'estudiante',
            'Activo'
        )
        ON CONFLICT (identificacion, tipodoc) DO UPDATE 
        SET nombre = EXCLUDED.nombre
        RETURNING id INTO v_user_id;

        -- 6. Sincronización en TESTUDIANTES (Ficha académica)

        VALUES (
            v_user_id,
            r->>'nombre',

FUNCTION,procesar_facturacion_solicitud,p_solicitud_id bigint,"CREATE OR REPLACE FUNCTION public.procesar_facturacion_solicitud(p_solicitud_id bigint)
 RETURNS jsonb
 LANGUAGE plpgsql
 SECURITY DEFINER
 SET search_path TO 'public'
AS $function$
DECLARE
    v_solicitud RECORD;
    v_precio RECORD;
    v_cantidad_premarcados INTEGER;
    v_cantidad_completaron INTEGER;
    v_precio_unitario NUMERIC(10, 2);
    v_subtotal NUMERIC(12, 2);
    v_descuento NUMERIC(12, 2);
    v_impuesto NUMERIC(12, 2);
    v_total NUMERIC(12, 2);
    v_numero_factura CHARACTER VARYING;
    v_factura_id BIGINT;
    v_resultado JSONB;

    -- Variables para el cálculo estimado
    v_subtotal_estimado NUMERIC(12, 2);
    v_descuento_estimado NUMERIC(12, 2);
    v_impuesto_estimado NUMERIC(12, 2);
    v_total_estimado NUMERIC(12, 2);
BEGIN
    
    SELECT 
        s.id, 
        c.id_dist AS id_distribuidor,
        s.id_colegio, 
        s.estado,
        s.id_grupo,
        s.fecha_inicio,
        s.fecha_fin
    INTO v_solicitud
    FROM public.tsolicitudes_aplicacion s
    JOIN public.tcolegios c ON s.id_colegio = c.id
    WHERE s.id = p_solicitud_id
    LIMIT 1;
    
    IF v_solicitud IS NULL THEN
        RETURN jsonb_build_object(
            'status', 'error',
            'message', 'Solicitud no encontrada',
            'solicitud_id', p_solicitud_id
        );
    END IF;
    
    IF v_solicitud.estado != 'aprobada' THEN
        RETURN jsonb_build_object(
            'status', 'error',
            'message', 'Solo se pueden facturar solicitudes aprobadas',
            'estado_actual', v_solicitud.estado,
            'solicitud_id', p_solicitud_id
        );
    END IF;
    
    -- Contar estudiantes premarcados (en el grupo)
    SELECT COUNT(*) INTO v_cantidad_premarcados
    FROM public.testudiantes
    WHERE id_grupo = v_solicitud.id_grupo;
    
    -- Contar estudiantes que completaron (tienen resultado)
    SELECT COUNT(*) INTO v_cantidad_completaron
    FROM public.tcompletados_pruebas
    WHERE id_solicitud = p_solicitud_id;
    
    IF v_cantidad_premarcados = 0 THEN
        -- No es un error crítico, puede que no haya estudiantes aún o que la solicitud sea para un grupo vacío
        -- Se permite continuar, pero se registra una advertencia.
        RAISE WARNING 'No hay estudiantes premarcados en el grupo % de la solicitud %', v_solicitud.id_grupo, p_solicitud_id;
    END IF;
    
    SELECT 
        id, 
        precio_base, 
        precio_por_estudiante, 
        descuento_porcentaje,
        impuesto_porcentaje
    INTO v_precio
    FROM public.tprecios
    WHERE estado = 'Activo'
        AND fecha_vigencia_desde <= CURRENT_DATE
        AND (fecha_vigencia_hasta IS NULL OR fecha_vigencia_hasta >= CURRENT_DATE)
    ORDER BY created_at DESC
    LIMIT 1;
    
    IF v_precio IS NULL THEN
        RETURN jsonb_build_object(
            'status', 'error',
            'message', 'No hay configuración de precios vigente',
            'fecha_actual', CURRENT_DATE
        );
    END IF;
    
    -- Cálculos para la factura real (basado en completados)
    v_precio_unitario := v_precio.precio_por_estudiante;
    v_subtotal := ROUND(v_precio.precio_base + (v_precio_unitario * v_cantidad_completaron), 2);
    v_descuento := ROUND(v_subtotal * (v_precio.descuento_porcentaje / 100), 2);
    v_impuesto := ROUND((v_subtotal - v_descuento) * (v_precio.impuesto_porcentaje / 100), 2);
    v_total := ROUND((v_subtotal - v_descuento) + v_impuesto, 2);

    -- Cálculos para el total estimado (basado en premarcados)
    v_subtotal_estimado := ROUND(v_precio.precio_base + (v_precio_unitario * v_cantidad_premarcados), 2);
    v_descuento_estimado := ROUND(v_subtotal_estimado * (v_precio.descuento_porcentaje / 100), 2);
    v_impuesto_estimado := ROUND((v_subtotal_estimado - v_descuento_estimado) * (v_precio.impuesto_porcentaje / 100), 2);
    v_total_estimado := ROUND((v_subtotal_estimado - v_descuento_estimado) + v_impuesto_estimado, 2);
    
    v_numero_factura := 'FAC-' || TO_CHAR(CURRENT_DATE, 'YYYY') || '-' || 
                       LPAD((SELECT COALESCE(MAX(CAST(SUBSTRING(numero_factura FROM 9) AS INTEGER)), 0) + 1 
                             FROM public.tfacturas WHERE numero_factura LIKE 'FAC-' || TO_CHAR(CURRENT_DATE, 'YYYY') || '-%')::TEXT, 6, '0');
    
    INSERT INTO public.tfacturas (
        id_distribuidor,
        id_colegio,
        id_solicitud_origen,
        numero_factura,
        fecha_emision,
        fecha_vencimiento,
        -- Premarcados y completaron
        cantidad_estudiantes_premarcados,
        cantidad_estudiantes_completaron,
        -- Componentes de precio (snapshot)
        precio_base_at_emission,
        descuento_porcentaje_at_emission,
        impuesto_porcentaje_at_emission,
        -- Calculados
        precio_unitario_calculado,
        subtotal_calculado,
        descuento_calculado,
        impuesto_calculado,
        total_calculado,
        -- Inicialmente real = calculado
        cantidad_estudiantes_real,
        precio_unitario_real,
        subtotal_real,
        descuento_real,
        impuesto_real,
        total_real,
        -- Finales (estos campos se usarán para el cierre de evaluación)
        subtotal,
        descuento,
        impuesto,
        total,
        cantidad_estudiantes,
        total_estimado,
        descripcion,
        estado
    )
    VALUES (
        v_solicitud.id_distribuidor,
        v_solicitud.id_colegio,
        p_solicitud_id,
        v_numero_factura,
        CURRENT_DATE,
        CURRENT_DATE + INTERVAL '30 days',
        v_cantidad_premarcados,
        v_cantidad_completaron,
        -- Componentes de precio (snapshot)
        v_precio.precio_base,
        v_precio.descuento_porcentaje,
        v_precio.impuesto_porcentaje,
        -- Calculados
        v_precio_unitario,
        v_subtotal,
        v_descuento,
        v_impuesto,
        v_total,
        v_cantidad_completaron,
        v_precio_unitario,
        v_subtotal,
        v_descuento,
        v_impuesto,
        v_total,
        v_subtotal,
        v_descuento,
        v_impuesto,
        v_total,
        v_cantidad_completaron,
        v_total_estimado,
        'Prueba vocacional - Premarcados: ' || v_cantidad_premarcados || ', Completaron: ' || v_cantidad_completaron || ', Período: ' || 
        TO_CHAR(v_solicitud.fecha_inicio, 'DD/MM/YYYY') || ' a ' || TO_CHAR(v_solicitud.fecha_fin, 'DD/MM/YYYY'),
        'Emitida'
    )
    RETURNING id INTO v_factura_id;
    
    v_resultado := jsonb_build_object(
        'status', 'success',
        'message', 'Factura generada correctamente',
        'factura_id', v_factura_id,
        'numero_factura', v_numero_factura,
        'solicitud_id', p_solicitud_id,
        'cantidad_premarcados', v_cantidad_premarcados,
        'cantidad_completaron', v_cantidad_completaron,
        'subtotal', v_subtotal,
        'descuento', v_descuento,
        'impuesto', v_impuesto,
        'total', v_total,
        'total_estimado', v_total_estimado,
        'fecha_emision', CURRENT_DATE,
        'fecha_vencimiento', CURRENT_DATE + INTERVAL '30 days'
    );
    
    RETURN v_resultado;
    
EXCEPTION WHEN OTHERS THEN
    RETURN jsonb_build_object(
        'status', 'error',
        'message', 'Error durante procesamiento de facturación',
        'error_detail', SQLERRM,
        'error_code', SQLSTATE
    );
END;

FUNCTION,trigger_registrar_completado,,"CREATE OR REPLACE FUNCTION public.trigger_registrar_completado()
 RETURNS trigger
 LANGUAGE plpgsql
 SECURITY DEFINER
 SET search_path TO 'public'
AS $function$
DECLARE
    v_id_solicitud BIGINT;
BEGIN
    -- CASO A: El estudiante finaliza la prueba (O se marca manualmente como Finalizado)
    IF (NEW.estado = 'Finalizado') THEN
        -- Localizamos la solicitud de aplicación aprobada
        SELECT sa.id INTO v_id_solicitud
        FROM public.tsolicitudes_aplicacion sa
        JOIN public.tgrupos g ON g.id = sa.id_grupo
        JOIN public.testudiantes e ON e.id_grupo = g.id
        WHERE e.id = NEW.id_estudiante
          AND sa.id_prueba = NEW.id_prueba
          AND sa.estado = 'aprobada'
        LIMIT 1;

        IF v_id_solicitud IS NOT NULL THEN
            INSERT INTO public.tcompletados_pruebas (
                id_solicitud,
                id_estudiante,
                id_resultado,
                timestamp_completado
            ) VALUES (
                v_id_solicitud,
                NEW.id_estudiante,
                NEW.id,
                NOW()
            )
            ON CONFLICT (id_solicitud, id_estudiante) DO NOTHING;
        END IF;

    -- CASO B: El test vuelve a estar en 'Proceso' (Reinicio administrativo)
    -- Si el registro existía en facturación, lo borramos inmediatamente.
    ELSIF (NEW.estado = 'Proceso') THEN
        DELETE FROM public.tcompletados_pruebas 
        WHERE id_estudiante = NEW.id_estudiante 
          AND id_resultado = NEW.id;
    END IF;

    RETURN NEW;
END;
$function$
